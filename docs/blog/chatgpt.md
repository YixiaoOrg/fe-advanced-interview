####  #怎么用js写一个throttle <span class="gpt-item">webxmsj@13:54:26 (还有4次)</span>

<pre>函数
/**
* throttle函数用于函数节流，即每隔一段时间段，只让该函数执行一次
* @param {function} fn 需要节流的函数
* @param {number} wait 函数节流的时长
*/
function throttle(fn, wait) {
    //上一次执行函数的时间
    let previous = 0;
    // 将throttle处理结果当作函数返回
    return function () {
        // 保留调用时的this上下文
        let context = this;
        // 保留调用时传入的参数
        let args = arguments;
        // 记录本次触发回调的时间
        let now = +new Date();
        // 判断上次触发的时间和本次触发的时间差是否小于时间间隔的阈值
        if (now - previous &gt; wait) {
            // 如果时间间隔大于我们设定的时间间隔阈值，则执行回调
            fn.apply(context, args);
            // 更新上一次回调触发的时间
            previous = now;
        }
    }
}</pre>
